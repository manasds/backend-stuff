Project Setup & Drizzle Schema

Lesson 1 · Recipe Manager API

What we're building

A Recipe Manager API with these entities:

  • Recipes — title, description, prep/cook time, difficulty
  • Ingredients — name, base unit
  • RecipeIngredients (join table with quantity) — many-to-many
  • Categories & Tags — many-to-many via join tables

Step 3: Drizzle schema

Create src/db/schema.ts — the heart of our data model:

import { pgTable, serial, text, integer, pgEnum, timestamp, uniqueIndex } from 'drizzle-orm/pg-core';
import { relations } from 'drizzle-orm';

// --- Enums ---
export const difficultyEnum = pgEnum('difficulty', ['easy', 'medium', 'hard']);

// --- Recipes ---
export const recipes = pgTable('recipes', {
  id: serial('id').primaryKey(),
  title: text('title').notNull(),
  description: text('description'),
  prepTime: integer('prep_time'),       // minutes
  cookTime: integer('cook_time'),       // minutes
  difficulty: difficultyEnum('difficulty').default('medium'),
  createdAt: timestamp('created_at').defaultNow().notNull(),
  updatedAt: timestamp('updated_at').defaultNow().notNull(),
});

// --- Ingredients ---
export const ingredients = pgTable('ingredients', {
  id: serial('id').primaryKey(),
  name: text('name').notNull().unique(),
  baseUnit: text('base_unit'),  // e.g. 'g', 'ml', 'tbsp', 'piece'
});

// --- Recipe <-> Ingredient (join table with quantity) ---
export const recipeIngredients = pgTable('recipe_ingredients', {
  id: serial('id').primaryKey(),
  recipeId: integer('recipe_id').notNull().references(() => recipes.id, { onDelete: 'cascade' }),
  ingredientId: integer('ingredient_id').notNull().references(() => ingredients.id, { onDelete: 'cascade' }),
  quantity: integer('quantity').notNull(),
  unit: text('unit'),  // optional override of ingredient's baseUnit
  notes: text('notes'),
});

// --- Categories ---
export const categories = pgTable('categories', {
  id: serial('id').primaryKey(),
  name: text('name').notNull().unique(),
  slug: text('slug').notNull().unique(),
});

// --- Recipe <-> Category ---
export const recipeCategories = pgTable('recipe_categories', {
  recipeId: integer('recipe_id').notNull().references(() => recipes.id, { onDelete: 'cascade' }),
  categoryId: integer('category_id').notNull().references(() => categories.id, { onDelete: 'cascade' }),
}, (t) => ({
  pk: uniqueIndex('recipe_category_pk').on(t.recipeId, t.categoryId),
}));

// --- Tags ---
export const tags = pgTable('tags', {
  id: serial('id').primaryKey(),
  name: text('name').notNull().unique(),
  slug: text('slug').notNull().unique(),
});

// --- Recipe <-> Tag ---
export const recipeTags = pgTable('recipe_tags', {
  recipeId: integer('recipe_id').notNull().references(() => recipes.id, { onDelete: 'cascade' }),
  tagId: integer('tag_id').notNull().references(() => tags.id, { onDelete: 'cascade' }),
}, (t) => ({
  pk: uniqueIndex('recipe_tag_pk').on(t.recipeId, t.tagId),
}));

Step 4: Schema relations

Create src/db/relations.ts:

import { relations } from 'drizzle-orm';
import { recipes, ingredients, recipeIngredients, categories, recipeCategories, tags, recipeTags } from './schema';

export const recipesRelations = relations(recipes, ({ many }) => ({
  recipeIngredients: many(recipeIngredients),
  recipeCategories: many(recipeCategories),
  recipeTags: many(recipeTags),
}));

export const ingredientsRelations = relations(ingredients, ({ many }) => ({
  recipeIngredients: many(recipeIngredients),
}));

export const recipeIngredientsRelations = relations(recipeIngredients, ({ one }) => ({
  recipe: one(recipes, { fields: [recipeIngredients.recipeId], references: [recipes.id] }),
  ingredient: one(ingredients, { fields: [recipeIngredients.ingredientId], references: [ingredients.id] }),
}));

export const categoriesRelations = relations(categories, ({ many }) => ({
  recipeCategories: many(recipeCategories),
}));

export const recipeCategoriesRelations = relations(recipeCategories, ({ one }) => ({
  recipe: one(recipes, { fields: [recipeCategories.recipeId], references: [recipes.id] }),
  category: one(categories, { fields: [recipeCategories.categoryId], references: [categories.id] }),
}));

export const tagsRelations = relations(tags, ({ many }) => ({
  recipeTags: many(recipeTags),
}));

export const recipeTagsRelations = relations(recipeTags, ({ one }) => ({
  recipe: one(recipes, { fields: [recipeTags.recipeId], references: [recipes.id] }),
  tag: one(tags, { fields: [recipeTags.tagId], references: [tags.id] }),
}));
Key insight: Drizzle relations are query helpers, not database constraints. Foreign keys are defined in references() on the table columns. Relations just tell Drizzle how to join when you use findMany() with with.

Step 5: DB connection

Create src/db/index.ts:

import { drizzle } from 'drizzle-orm/postgres-js';
import postgres from 'postgres';
import * as schema from './schema';
import * as relations from './relations';

const connectionString = process.env.DATABASE_URL!;
const client = postgres(connectionString);
export const db = drizzle(client, { schema: { ...schema, ...relations } });

Step 6: Drizzle config & migrate

Create drizzle.config.ts:

import { defineConfig } from 'drizzle-kit';

export default defineConfig({
  schema: './src/db/schema.ts',
  out: './drizzle',
  dialect: 'postgresql',
  dbCredentials: { url: process.env.DATABASE_URL! },
});
$ npx drizzle-kit generate
$ npx drizzle-kit migrate

What you learned

  • Many-to-many relationships via join tables with extra columns (quantity)
  • Drizzle relations for query-time joins
  • Composite unique indexes on join tables
  • Cascade deletes on foreign keys
